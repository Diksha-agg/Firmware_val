1function [F_traj, M_traj] = run_sim()
 2max_time=4;
 3tstep    = 0.01; % this determines the time step at which the solution is given
 4max_iter = max_time/tstep; % max iteration
 5time=0;
 6des_z=zeros(max_iter,1);
 7curr_z=zeros(max_iter,1);
 8ttraj=zeros(max_iter,1);
 9F_traj=zeros(max_iter,1);
10M_traj=zeros(max_iter,3);
11for iter=1:max_iter
12    t=time;
13    [posd, veld, rotd, omegad, controld] = time_trajj(t);
14    [posc, velc, rotc, omegac] = local(t);
15    [F,M] = controller(posc, velc, rotc, omegac ,posd, veld, rotd, omegad, controld);
16    des_z(iter)=posd(3);
17    curr_z(iter)=posc(3);
18    F_traj(iter)=F;
19    M_traj(iter,:)=M;
20    ttraj(iter)=time;
21    time=t+tstep;    
22end
23figure(1)
24plot(ttraj,des_z,'r');
25hold on
26plot(ttraj,curr_z,'b');
27ylabel('z')
28disp('Simulation Finished.');
29end
