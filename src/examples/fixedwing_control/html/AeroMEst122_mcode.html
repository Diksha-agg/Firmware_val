1function [ Moment_aero ] = AeroMEst( eul, x_dot,omega, Fa);
 2%UNTITLED2 Summary of this function goes here
 3%   Detailed explanation goes here
 4coder.inline('never');
 5BQ_x_ac = [0.0838;0;0];
 6BQ_x_cg = [0.157;0;0];
 7c_root=0.39;
 8c_tip=0.176;
 9t= c_tip/c_root;
10BQ_c = c_root*(2/3)*((1+t+t^2)/(1+t));
11BQ_b=2.29/2;
12BQ_S=0.754/2;
13
14BQ_Cm_0=-0.0156;
15BQ_Cm_alpha=0.995;
16BQ_Cm_q=-0.51;
17BQ_Cl_beta=0;
18BQ_Cl_p=-0.43;
19BQ_Cl_r=0.29;
20BQ_Cn_beta=0.0812;
21BQ_Cn_p=-0.4044;
22BQ_Cn_r=-0.05;
23
24rho=1.225;
25Moment_aero=[0;0;0];
26
27R=eul2rotm(eul);
28xb_dot=R'*x_dot;
29V=norm(x_dot);
30
31if (V==0) return;
32end
33
34Rq2w = [
35    0 0 1;
36    0 1 0;
37    -1 0 0];
38xw_dot = Rq2w*xb_dot;
39
40if (xw_dot(1)~=0)
41    alpha=atan2(-xw_dot(3),xw_dot(1));
42    beta=atan2(xw_dot(2),xw_dot(1));
43else
44    alpha=0;
45    beta=0;
46end
47
48omega_w = Rq2w*omega;
49p_w = omega_w(1,1);
50q_w = omega_w(2,1);
51r_w = omega_w(3,1);
52
53C_l = BQ_Cl_beta*beta + BQ_Cl_p*p_w*BQ_b/(2*V) + BQ_Cl_r*r_w*BQ_b/(2*V);
54C_m = BQ_Cm_0 + BQ_Cm_alpha*alpha + BQ_Cm_q*q_w*BQ_c/(2*V);
55C_n = BQ_Cn_beta*beta + BQ_Cn_p*p_w*BQ_b/(2*V) + BQ_Cn_r*r_w*BQ_b/(2*V);
56
57Mx_w_ac = 0.5*rho*(V^2)*BQ_S*BQ_c*C_l ;
58My_w_ac = 0.5*rho*(V^2)*BQ_S*BQ_c*C_m ;
59Mz_w_ac = 0.5*rho*(V^2)*BQ_S*BQ_c*C_n ;
60
61M_ac = [Mx_w_ac;My_w_ac;Mz_w_ac];% wing frame
62% r = BQ.x_ac - BQ.x_cg;
63% M_cg = M_ac + cross(r',(Rq2w*Fa)')';
64r = BQ_x_ac(1) - BQ_x_cg(1);
65% L = -Fa(1);
66% M_cg = M_ac + [0;r*L;0]; % wing frame
67
68% Moment_aero = BQ.wing_n*Rq2w'*M_cg; % body frame
69
70Moment_aero = [0;2*My_w_ac+r*Fa(1);0];
71
72end
