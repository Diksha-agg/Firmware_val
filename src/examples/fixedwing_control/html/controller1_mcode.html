1function [M, Thrust] = controller(posc, velc, rotc, omegac ,t)
 2coder.inline('never');
 3%desired inputs:[xd yd zd xd_dot yd_dot zd_dot phid thetad psid phidotd thetadotd psidotd ud(1)---Tfwd
 4%ud(2)---Mfwd
 5[posd, veld, rot_des, omegad, controld] = time_trajj(t);
 6Thrust=0;
 7BQ_m=12;%kg
 8BQ_g=9.81;
 9BQ_J=[1.86 0 0;
10    0 2.031 0;
11    0 0 3.617];
12ud=controld;
13
14omega_curr=[1 0 -sin(rotc(2));0 cos(rotc(3)) cos(rotc(2))*sin(rotc(3));0 -sin(rotc(3)) cos(rotc(2))*cos(rotc(3))]*[omegac(3); omegac(2); omegac(1)];
15omega_des=[1 0 -sin(rot_des(2));0 cos(rot_des(3)) cos(rot_des(2))*sin(rot_des(3));0 -sin(rot_des(3)) cos(rot_des(2))*cos(rot_des(3))]*[omegad(3); omegad(2); omegad(1)];
16
17kp=eye(3);
18kp(1,1)=5;
19kp(2,2)=5;
20kp(3,3)=50;
21
22kv=eye(3);
23kv(1,1)=50;     
24kv(2,2)=6;
25kv(3,3)=5;
26
27kr=eye(3);
28kr(1,1)=3;
29kr(2,2)=100;
30kr(3,3)=5;
31
32kw=eye(3);
33kw(1,1)=5;
34kw(2,2)=1500;
35kw(3,3)=1;
36
37Rb=eul2rotm(rotc);
38
39R=eul2rotm(rot_des);
40Fa=AeroFEst([0 rotc(2) 0],[velc(1);0;velc(3)],omega_curr);
41acc_net=((R*[0;0;ud(1)])/BQ_m)+kp*[0;0;(posd(3)-posc(3))]+kv*[(veld(1)-velc(1));-velc(2);(veld(3)-velc(3))]+[0; 0; BQ_g]-((Rb*Fa)/(BQ_m)); % acc_net
42
43% calculation of current euler angles
44b3=acc_net/norm(acc_net);
45c2=[-sin(rot_des(1)) cos(rot_des(1)) 0]';
46b1=cross(c2,b3)/norm(cross(c2,b3));
47b2=cross(b3,b1);
48Rd=[b1 b2 b3];
49nm=norm(acc_net);
50%thrust control input
51%flag=sign([0 0 1]*Rb'*acc_net);
52Thrust=(BQ_m*nm);           
53
54R=eul2rotm(rotc);
55erm=0.5*((Rd'*R)-(R'*Rd));
56er=[erm(3,2);erm(1,3);erm(2,1)];
57ew=(omega_curr-((R'*Rd)*omega_des));
58
59tau_a=AeroMEst([0; rotc(2); 0],[velc(1);0;velc(3)],omega_curr, Fa);
60M=[0;ud(2);0]-kr*er-kw*ew+cross(omega_curr,BQ_J*omega_curr)-BQ_J*cross(ew,(R'*Rd)*omega_des)-tau_a;    %moment input
61
62end
